# YYC³ CloudPivot Intelli-Matrix - 配置管理改进建议

**分析时间**: 2026-03-26
**问题**: 配置文件分散，操作性不高，备份不便

---

## 🔴 当前问题分析

### 问题一：配置文件分散

| 文件位置 | 文件名 | 用途 | 问题 |
|----------|--------|------|------|
| `/` | `env-config.ts` | 环境变量配置 | ❌ 与 src/app/lib/env-config.ts 重复 |
| `/` | `.env.example` | 环境变量示例 | ❌ 不在 src 目录，备份时丢失 |
| `/src/app/lib/` | `env-config.ts` | 环境变量配置 | ⚠️ 与根目录重复 |
| `/src/app/lib/` | `api-config.ts` | API 端点配置 | ✅ 位置正确 |
| `/src/app/lib/` | `ollama-url.ts` | Ollama 端点解析 | ✅ 位置正确 |

### 问题二：配置重复

**重复的配置文件**:
- `/env-config.ts` (226 行)
- `/src/app/lib/env-config.ts` (226 行)

**重复的配置内容**:
- 系统标识 (SYSTEM_NAME, SYSTEM_VERSION, SYSTEM_BUILD)
- 网络端点 (API_BASE_URL, WS_ENDPOINT, OLLAMA_BASE_URL)
- 存储配置 (STORAGE_PREFIX, IDB_NAME, IDB_VERSION)
- AI 配置 (DEFAULT_AI_BASE_URL, DEFAULT_AI_MODEL, ...)
- 功能开关 (ENABLE_MOCK_MODE, ENABLE_DEBUG, ENABLE_PWA, ...)
- 安全配置 (SESSION_TIMEOUT_MIN, MAX_LOGIN_ATTEMPTS, CORS_ORIGINS)

### 问题三：备份不便

**用户备份策略**: 只备份 `src` 目录

**问题**:
- ❌ 根目录的 `env-config.ts` 不会被备份
- ❌ 根目录的 `.env.example` 不会被备份
- ❌ 如果依赖根目录配置，恢复后配置丢失
- ❌ 需要单独备份配置文件

### 问题四：操作性不高

**配置修改复杂**:
- 需要同时修改多个配置文件
- 配置分散在不同位置
- 难以统一管理
- 容易出现配置不一致

**环境切换困难**:
- 开发/生产环境配置切换需要修改多处
- 缺少统一的环境配置管理
- 没有配置验证机制

---

**目标**: 所有配置文件都在 `src` 目录内，备份 `src` 时配置也能一起备份。

```typescript
// src/config/env-config.ts
/**
 * 环境变量配置
 * ============

import { getSharedChannel } from "../app/lib/broadcast-channel";

export interface EnvConfig {
  // 系统标识
  SYSTEM_NAME: string;
  SYSTEM_VERSION: string;
  SYSTEM_BUILD: string;

  // 网络端点
  API_BASE_URL: string;
  WS_ENDPOINT: string;
  OLLAMA_BASE_URL: string;
  OLLAMA_PROXY_PATH: string;

  // 存储配置
  STORAGE_PREFIX: string;
  IDB_NAME: string;
  IDB_VERSION: number;

  // 集群标识
  CLUSTER_ID: string;
  NODE_ENV: string;

  // AI 默认配置
  DEFAULT_AI_BASE_URL: string;
  DEFAULT_AI_MODEL: string;
  DEFAULT_AI_TEMPERATURE: number;
  DEFAULT_AI_MAX_TOKENS: number;
  DEFAULT_AI_TIMEOUT: number;

  // 安全配置
  SESSION_TIMEOUT_MIN: number;
  MAX_LOGIN_ATTEMPTS: number;
  CORS_ORIGINS: string;

  // 功能开关
  ENABLE_MOCK_MODE: boolean;
  ENABLE_DEBUG: boolean;
  ENABLE_PWA: boolean;

  STORAGE_PREFIX: "yyc3_",
  IDB_NAME: "yyc3_matrix",
  IDB_VERSION: 3,

  CLUSTER_ID: "CN-EAST-PROD-01",
  NODE_ENV: "development",

  DEFAULT_AI_BASE_URL: "https://api.openai.com/v1",
  DEFAULT_AI_MODEL: "gpt-4o",
  DEFAULT_AI_TEMPERATURE: 0.7,
  DEFAULT_AI_MAX_TOKENS: 2048,

  MAX_LOGIN_ATTEMPTS: 5,

  ENABLE_MOCK_MODE: true,
  ENABLE_DEBUG: false,
  ENABLE_PWA: true,

const STORAGE_KEY = "yyc3_env_config";
const CONFIG_CHANNEL_NAME = "yyc3_env_config";

let _envConfig: EnvConfig | null = null;

function loadEnvOverrides(): Partial<EnvConfig> {
  const overrides: Partial<EnvConfig> = {};
  try {
    const metaEnv = (import.meta as unknown as Record<string, unknown>).env as Record<string, string> | undefined;
    if (metaEnv) {
      if (metaEnv.VITE_YYC3_SYSTEM_NAME)     {overrides.SYSTEM_NAME = metaEnv.VITE_YYC3_SYSTEM_NAME;}
      if (metaEnv.VITE_YYC3_SYSTEM_VERSION)   {overrides.SYSTEM_VERSION = metaEnv.VITE_YYC3_SYSTEM_VERSION;}
      if (metaEnv.VITE_YYC3_API_BASE_URL)     {overrides.API_BASE_URL = metaEnv.VITE_YYC3_API_BASE_URL;}
      if (metaEnv.VITE_YYC3_WS_ENDPOINT)      {overrides.WS_ENDPOINT = metaEnv.VITE_YYC3_WS_ENDPOINT;}
      if (metaEnv.VITE_YYC3_OLLAMA_BASE_URL)  {overrides.OLLAMA_BASE_URL = metaEnv.VITE_YYC3_OLLAMA_BASE_URL;}
      if (metaEnv.VITE_YYC3_OLLAMA_PROXY_PATH) {overrides.OLLAMA_PROXY_PATH = metaEnv.VITE_YYC3_OLLAMA_PROXY_PATH;}
      if (metaEnv.VITE_YYC3_CLUSTER_ID)       {overrides.CLUSTER_ID = metaEnv.VITE_YYC3_CLUSTER_ID;}
      if (metaEnv.VITE_YYC3_STORAGE_PREFIX)    {overrides.STORAGE_PREFIX = metaEnv.VITE_YYC3_STORAGE_PREFIX;}
      if (metaEnv.VITE_YYC3_ENABLE_MOCK)       {overrides.ENABLE_MOCK_MODE = metaEnv.VITE_YYC3_ENABLE_MOCK === "true";}
      if (metaEnv.VITE_YYC3_ENABLE_DEBUG)      {overrides.ENABLE_DEBUG = metaEnv.VITE_YYC3_ENABLE_DEBUG === "true";}
      if (metaEnv.MODE)                        {overrides.NODE_ENV = metaEnv.MODE;}
    }
  } catch { /* ignore */ }
  return overrides;
}

function loadStoredConfig(): Partial<EnvConfig> {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    if (raw) {return JSON.parse(raw);}
  } catch { /* ignore */ }
  return {};
}

function buildConfig(): EnvConfig {
  const envOverrides = loadEnvOverrides();
  const stored = loadStoredConfig();
  return { ...DEFAULTS, ...stored, ...envOverrides };
}

export function env<K extends keyof EnvConfig>(key: K): EnvConfig[K] {
  if (!_envConfig) {_envConfig = buildConfig();}
  return _envConfig[key];
}

export function getEnvConfig(): Readonly<EnvConfig> {
  if (!_envConfig) {_envConfig = buildConfig();}
  return { ..._envConfig };
}

export function setEnvConfig(updates: Partial<EnvConfig>): EnvConfig {
  if (!_envConfig) {_envConfig = buildConfig();}
  _envConfig = { ..._envConfig, ...updates };
  try {
    localStorage.setItem(STORAGE_KEY, JSON.stringify(_envConfig));
  } catch { /* ignore */ }

  // 广播到其他标签页
  try {
    const ch = getSharedChannel(CONFIG_CHANNEL_NAME);
    ch?.postMessage({ type: "config_update", config: _envConfig });
  } catch { /* ignore */ }

  return { ..._envConfig };
}

export function resetEnvConfig(): EnvConfig {
  try { localStorage.removeItem(STORAGE_KEY); } catch { /* ignore */ }
  _envConfig = null;
  return getEnvConfig();
}

export function exportEnvConfig(): string {
  return JSON.stringify({
    _type: "env-config",
    _exportedAt: new Date().toISOString(),
    config: getEnvConfig(),
  }, null, 2);
}

export function importEnvConfig(json: string): boolean {
  try {
    const parsed = JSON.parse(json);
    const config = parsed.config || parsed;
    setEnvConfig(config);
    return true;
  } catch {
    return false;
  }
}
```

```typescript
// src/config/preset-manager.ts
import { setEnvConfig, getEnvConfig } from "./env-config";
import { developmentPreset } from "./presets/development";
import { stagingPreset } from "./presets/staging";
import { productionPreset } from "./presets/production";
import { testPreset } from "./presets/test";

export type Environment = "development" | "staging" | "production" | "test";

const PRESETS: Record<Environment, Partial<EnvConfig>> = {
  development: developmentPreset,
  staging: stagingPreset,
  production: productionPreset,
  test: testPreset,
};

export function switchEnvironment(env: Environment): void {
  const preset = PRESETS[env];
  if (preset) {
    setEnvConfig(preset);
    console.log(`[Config] Switched to ${env} environment`);
  }
}

export function getCurrentEnvironment(): Environment {
  const config = getEnvConfig();
  return config.NODE_ENV as Environment;
}

export function exportEnvironmentPreset(env: Environment): string {
  const preset = PRESETS[env];
  return JSON.stringify(preset, null, 2);
}
```

### 方案三：配置验证工具（可选）

**目标**: 提供配置验证功能，确保配置正确性。

```typescript
// src/config/validator.ts
import type { EnvConfig } from "./env-config";
import { getEnvConfig } from "./env-config";

export interface ValidationResult {
  valid: boolean;
  errors: string[];
  warnings: string[];
}

export function validateConfig(config: EnvConfig): ValidationResult {
  const errors: string[] = [];
  const warnings: string[] = [];

  // 验证 URL 格式
  try {
    new URL(config.API_BASE_URL);
  } catch {
    errors.push("API_BASE_URL is not a valid URL");
  }

  try {
    new URL(config.WS_ENDPOINT);
  } catch {
    errors.push("WS_ENDPOINT is not a valid WebSocket URL");
  }

  try {
    new URL(config.OLLAMA_BASE_URL);
  } catch {
    errors.push("OLLAMA_BASE_URL is not a valid URL");
  }

  // 验证数值范围
  if (config.SESSION_TIMEOUT_MIN < 5 || config.SESSION_TIMEOUT_MIN > 120) {
    warnings.push("SESSION_TIMEOUT_MIN should be between 5 and 120");
  }

  if (config.DB_POOL_MIN < 1 || config.DB_POOL_MIN > 10) {
    warnings.push("DB_POOL_MIN should be between 1 and 10");
  }

  if (config.DB_POOL_MAX < config.DB_POOL_MIN) {
    errors.push("DB_POOL_MAX should be greater than DB_POOL_MIN");
  }

  // 验证存储前缀
  if (!/^[a-z0-9_]+$/.test(config.STORAGE_PREFIX)) {
    errors.push("STORAGE_PREFIX should only contain lowercase letters, numbers, and underscores");
  }

  return {
    valid: errors.length === 0,
    errors,
    warnings,
  };
}

export function validateCurrentConfig(): ValidationResult {
  const config = getEnvConfig();
  return validateConfig(config);
}
```
