```bash
# 智能文档检查（推荐）
pnpm run docs:smart-check

# 批量处理文档
pnpm run docs:batch:headers  # 添加标头

# 完整文档检查
pnpm run docs:check

# 查看详细报告
cat .archive/reports/smart-doc-check-*.json
```
