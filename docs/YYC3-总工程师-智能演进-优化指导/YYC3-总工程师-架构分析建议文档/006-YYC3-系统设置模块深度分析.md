# YYC³ 系统设置模块深度分析报告

## 一、模块概述

### 1.1 功能定位

系统设置模块是 YYC³ Cloud Intelli-Matrix 的核心配置管理中心，负责整个系统的参数配置、API 端点管理、模型提供商配置、安全设置、通知管理等功能。该模块采用分区域、分层次的配置架构，支持实时配置更新和跨标签页同步。

## 五、具体优化方案

### 5.1 短期优化 (1-2 周)

#### 5.1.1 配置验证增强

```typescript
// src/app/lib/config-validator.ts

import type { APIEndpoints } from "./api-config";

interface ValidationResult {
  valid: boolean;
  errors: ConfigError[];
  warnings: ConfigWarning[];
}

interface ConfigError {
  field: string;
  message: string;
  value: unknown;
}

interface ConfigWarning {
  field: string;
  message: string;
  suggestion: string;
}

const URL_PATTERN = /^https?:\/\/.+/;
const PORT_PATTERN = /^\d{1,5}$/;
const API_KEY_PATTERN = /^[\w-]{20,}$/;

export function validateAPIConfig(config: APIEndpoints): ValidationResult {
  const errors: ConfigError[] = [];
  const warnings: ConfigWarning[] = [];

  // 验证 API 端点 URL 格式
  if (config.baseURL && !URL_PATTERN.test(config.baseURL)) {
    errors.push({
      field: "baseURL",
      message: "无效的 URL 格式",
      value: config.baseURL,
    });
  }

  // 验证端口范围
  if (config.port) {
    const port = parseInt(config.port as string, 10);
    if (isNaN(port) || port < 1 || port > 65535) {
      errors.push({
        field: "port",
        message: "端口号必须在 1-65535 范围内",
        value: config.port,
      });
    }
  }

  // 验证超时配置
  if (config.timeout && config.timeout < 1000) {
    warnings.push({
      field: "timeout",
      message: "超时时间过短可能导致请求失败",
      suggestion: "建议设置至少 5000ms",
    });
  }

  // 验证重试次数
  if (config.retries && config.retries > 10) {
    warnings.push({
      field: "retries",
      message: "重试次数过多可能影响响应速度",
      suggestion: "建议设置 3-5 次",
    });
  }

  return {
    valid: errors.length === 0,
    errors,
    warnings,
  };
}

export function validateAPIKey(key: string, provider: string): ValidationResult {
  const errors: ConfigError[] = [];
  const warnings: ConfigWarning[] = [];

  if (!key) {
    errors.push({
      field: "apiKey",
      message: "API Key 不能为空",
      value: "",
    });
  } else if (!API_KEY_PATTERN.test(key)) {
    warnings.push({
      field: "apiKey",
      message: "API Key 格式可能不正确",
      suggestion: `请检查 ${provider} API Key 格式`,
    });
  }

  return {
    valid: errors.length === 0,
    errors,
    warnings,
  };
}
```

```typescript
    if (!version) return false;

    this.createVersion(`回滚前自动保存`);

    // 恢复历史配置
    setAPIConfig(version.config);
    return true;
  }

  // 对比两个版本
  compareVersions(v1Id: string, v2Id: string): ConfigDiff[] {
    const v1 = this.versions.find(v => v.id === v1Id);
    const v2 = this.versions.find(v => v.id === v2Id);
    
    if (!v1 || !v2) return [];

    const diffs: ConfigDiff[] = [];
    const keys = new Set([...Object.keys(v1.config), ...Object.keys(v2.config)]);

    keys.forEach(key => {
      const val1 = v1.config[key as keyof APIEndpoints];
      const val2 = v2.config[key as keyof APIEndpoints];
      
      if (val1 !== val2) {
        diffs.push({
          field: key,
          oldValue: val1,
          newValue: val2,
        });
      }
    });

    return diffs;
  }

  // 删除指定版本
  deleteVersion(versionId: string): boolean {
    const index = this.versions.findIndex(v => v.id === versionId);
    if (index === -1) return false;

    this.versions.splice(index, 1);
    this.save();
    return true;
  }
```

```typescript
import CryptoJS from "crypto-js";

const ENCRYPTION_KEY = "yyc3-config-encryption-key";

export class ConfigEncryption {
  // 加密敏感配置
  static encrypt(value: string): string {
    return CryptoJS.AES.encrypt(value, ENCRYPTION_KEY).toString();
  }

  // 解密敏感配置
  static decrypt(encrypted: string): string {
    try {
      const bytes = CryptoJS.AES.decrypt(encrypted, ENCRYPTION_KEY);
      return bytes.toString(CryptoJS.enc.Utf8);
    } catch {
      return "";
    }
  }

  // 加密对象
  static encryptObject(obj: Record<string, unknown>): string {
    const json = JSON.stringify(obj);
    return this.encrypt(json);
  }

  // 解密对象
  static decryptObject<T>(encrypted: string): T | null {
    try {
      const json = this.decrypt(encrypted);
      return JSON.parse(json) as T;
    } catch {
      return null;
    }
  }
}

// 敏感字段列表
const SENSITIVE_FIELDS = ["apiKey", "secretKey", "password", "token"];

export function encryptSensitiveConfig(config: Record<string, unknown>): Record<string, unknown> {
  const encrypted = { ...config };
  
  SENSITIVE_FIELDS.forEach(field => {
    if (encrypted[field] && typeof encrypted[field] === "string") {
      encrypted[field] = ConfigEncryption.encrypt(encrypted[field] as string);
    }
  });

  return encrypted;
}

export function decryptSensitiveConfig(config: Record<string, unknown>): Record<string, unknown> {
  const decrypted = { ...config };
  
  SENSITIVE_FIELDS.forEach(field => {
    if (decrypted[field] && typeof decrypted[field] === "string") {
      decrypted[field] = ConfigEncryption.decrypt(decrypted[field] as string);
    }
  });

  return decrypted;
}
```

```typescript
interface Environment {

  isDefault: boolean;

const ENVIRONMENTS_KEY = "yyc3_environments";
const CURRENT_ENV_KEY = "yyc3_current_environment";

export class EnvironmentManager {
  private environments: Environment[] = [];
  private currentEnvId: string = "default";

  constructor() {
    this.load();
  }

  private load() {
    try {
      const raw = localStorage.getItem(ENVIRONMENTS_KEY);
      this.environments = raw ? JSON.parse(raw) : [this.createDefaultEnv()];
      
      this.currentEnvId = localStorage.getItem(CURRENT_ENV_KEY) || "default";
    } catch {
      this.environments = [this.createDefaultEnv()];
    }
  }

  private save() {
    localStorage.setItem(ENVIRONMENTS_KEY, JSON.stringify(this.environments));
    localStorage.setItem(CURRENT_ENV_KEY, this.currentEnvId);
  }

  private createDefaultEnv(): Environment {
    return {
      id: "default",
      name: "默认环境",
      description: "系统默认配置",
      config: getAPIConfig(),
      isDefault: true,
      createdAt: Date.now(),
      updatedAt: Date.now(),
    };
  }

  // 创建新环境
  createEnvironment(name: string, description: string, baseConfig?: APIEndpoints): Environment {
    const env: Environment = {
      id: `env-${Date.now()}`,
      name,
      description,
      config: baseConfig || { ...getAPIConfig() },
      isDefault: false,
      createdAt: Date.now(),
      updatedAt: Date.now(),
    };

    this.environments.push(env);
    this.save();
    return env;
  }

  // 切换环境
  switchEnvironment(envId: string): boolean {
    const env = this.environments.find(e => e.id === envId);
    if (!env) return false;

    // 保存当前环境配置
    const currentEnv = this.getCurrentEnvironment();
    if (currentEnv) {
      currentEnv.config = getAPIConfig();
      currentEnv.updatedAt = Date.now();
    }

    // 切换到新环境
    this.currentEnvId = envId;
    setAPIConfig(env.config);
    this.save();
    return true;
  }

  // 获取当前环境
  getCurrentEnvironment(): Environment | undefined {
    return this.environments.find(e => e.id === this.currentEnvId);
  }

  // 获取所有环境
  getEnvironments(): Environment[] {
    return this.environments;
  }

  // 复制环境
  duplicateEnvironment(envId: string, newName: string): Environment | null {
    const env = this.environments.find(e => e.id === envId);
    if (!env) return null;

    return this.createEnvironment(newName, `${env.description} (副本)`, env.config);
  }

  // 删除环境
  deleteEnvironment(envId: string): boolean {
    if (envId === "default") return false;

    const index = this.environments.findIndex(e => e.id === envId);
    if (index === -1) return false;

    this.environments.splice(index, 1);
    
    // 如果删除的是当前环境，切换到默认环境
    if (this.currentEnvId === envId) {
      this.switchEnvironment("default");
    }

    this.save();
    return true;
  }

  // 导出环境配置
  exportEnvironment(envId: string): string {
    const env = this.environments.find(e => e.id === envId);
    if (!env) return "";

    return JSON.stringify(env, null, 2);
  }

  // 导入环境配置
  importEnvironment(json: string): Environment | null {
    try {
      const env = JSON.parse(json) as Environment;
      env.id = `env-${Date.now()}`;
      env.isDefault = false;
      env.createdAt = Date.now();
      env.updatedAt = Date.now();

      this.environments.push(env);
      this.save();
      return env;
    } catch {
      return null;
    }
  }
}
```

```typescript
interface AuditLog {
  id: string;
  timestamp: number;

}

const AUDIT_LOGS_KEY = "yyc3_config_audit_logs";
const MAX_LOGS = 1000;

export class ConfigAuditLogger {
  private logs: AuditLog[] = [];

  constructor() {
    this.load();
  }

  private load() {
    try {
      const raw = localStorage.getItem(AUDIT_LOGS_KEY);
      this.logs = raw ? JSON.parse(raw) : [];
    } catch {
      this.logs = [];
    }
  }

  private save() {
    localStorage.setItem(AUDIT_LOGS_KEY, JSON.stringify(this.logs));
  }

  // 记录配置变更
  log(
    action: AuditLog["action"],
    field: string,
    oldValue: unknown,
    newValue: unknown,
    metadata?: { userId?: string; userAgent?: string; ip?: string }
  ) {
    const log: AuditLog = {
      id: `audit-${Date.now()}`,
      timestamp: Date.now(),
      action,
      field,
      oldValue,
      newValue,
      ...metadata,
    };

    this.logs.unshift(log);

    // 保留最近 MAX_LOGS 条记录
    if (this.logs.length > MAX_LOGS) {
      this.logs = this.logs.slice(0, MAX_LOGS);
    }

    this.save();
  }

  // 获取审计日志
  getLogs(filter?: {
    action?: AuditLog["action"];
    field?: string;
    startTime?: number;
    endTime?: number;
  }): AuditLog[] {
    let filtered = [...this.logs];

    if (filter) {
      if (filter.action) {
        filtered = filtered.filter(l => l.action === filter.action);
      }
      if (filter.field) {
        filtered = filtered.filter(l => l.field.includes(filter.field!));
      }
      if (filter.startTime) {
        filtered = filtered.filter(l => l.timestamp >= filter.startTime!);
      }
      if (filter.endTime) {
        filtered = filtered.filter(l => l.timestamp <= filter.endTime!);
      }
    }

    return filtered;
  }

  // 获取变更统计
  getStatistics(): {
    totalChanges: number;
    byAction: Record<string, number>;
    byField: Record<string, number>;
    recentActivity: number;
  } {
    const byAction: Record<string, number> = {};
    const byField: Record<string, number> = {};

    this.logs.forEach(log => {
      byAction[log.action] = (byAction[log.action] || 0) + 1;
      byField[log.field] = (byField[log.field] || 0) + 1;
    });

    const oneDayAgo = Date.now() - 24 * 60 * 60 * 1000;
    const recentActivity = this.logs.filter(l => l.timestamp > oneDayAgo).length;

    return {
      totalChanges: this.logs.length,
      byAction,
      byField,
      recentActivity,
    };
  }

  // 清理旧日志
  clearOldLogs(daysToKeep: number = 30) {
    const cutoff = Date.now() - daysToKeep * 24 * 60 * 60 * 1000;
    this.logs = this.logs.filter(l => l.timestamp > cutoff);
    this.save();
  }
}
```
