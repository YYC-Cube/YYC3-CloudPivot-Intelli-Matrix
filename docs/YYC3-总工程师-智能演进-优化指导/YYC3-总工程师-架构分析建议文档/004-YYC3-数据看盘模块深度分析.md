# YYC³ 数据看盘模块深度分析报告

## 一、模块概述

### 1.1 功能定位

数据看盘（Dashboard）是 YYC³ Cloud Intelli-Matrix 的核心监控模块，承担着系统状态可视化、实时数据展示、异常告警提示等关键功能。该模块采用 **Recharts** 图表库实现数据可视化，通过 **WebSocket** 实时推送数据，支持桌面端和移动端自适应布局。

### 1.2 核心组件架构

```
┌─────────────────────────────────────────────────────────────────────────────────────────┐
│                              数据看盘模块架构                                            │
├─────────────────────────────────────────────────────────────────────────────────────────┤

│  ┌─────────────────────────────────────────────────────────────────────────────────┐   │
│  │  Dashboard.tsx (主组件)                                                          │   │
```

#### 5.1.3 图表错误边界

```typescript
// src/app/components/ChartErrorBoundary.tsx

import React, { Component, ErrorInfo, ReactNode } from "react";

interface Props {
  children: ReactNode;
  fallback?: ReactNode;
}

interface State {
  hasError: boolean;
  error?: Error;
}

export class ChartErrorBoundary extends Component<Props, State> {
  public state: State = { hasError: false };

  public static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error };
  }

  public componentDidCatch(error: Error, errorInfo: ErrorInfo) {
    console.error("Chart rendering error:", error, errorInfo);
  }

  public render() {
    if (this.state.hasError) {
      return this.props.fallback || (
        <div className="p-4 rounded-lg bg-[rgba(255,51,102,0.1)] border border-[rgba(255,51,102,0.2)]">
          <p className="text-[#ff3366]">图表加载失败，请刷新页面重试</p>
        </div>
      );
    }

    return this.props.children;
  }
}
```

```typescript
import { useCallback } from "react";

export function useChartExport() {
  const exportAsPNG = useCallback(async (elementId: string, filename: string) => {
    const element = document.getElementById(elementId);
    if (!element) return;

    const canvas = await html2canvas(element, {
      backgroundColor: "#060e1f",
      scale: 2,
    });

    const link = document.createElement("a");
    link.download = `${filename}.png`;
    link.href = canvas.toDataURL("image/png");
    link.click();
  }, []);

  const exportAsSVG = useCallback((svgElement: SVGSVGElement, filename: string) => {
    const serializer = new XMLSerializer();
    const svgString = serializer.serializeToString(svgElement);
    const blob = new Blob([svgString], { type: "image/svg+xml" });
    const url = URL.createObjectURL(blob);

    const link = document.createElement("a");
    link.download = `${filename}.svg`;
    link.href = url;
    link.click();

    URL.revokeObjectURL(url);
  }, []);

  const exportDataAsCSV = useCallback((data: Record<string, unknown>[], filename: string) => {
    const headers = Object.keys(data[0] || {});
    const csvContent = [
      headers.join(","),
      ...data.map(row => headers.map(h => row[h]).join(","))
    ].join("\n");

    const blob = new Blob([csvContent], { type: "text/csv;charset=utf-8;" });
    const url = URL.createObjectURL(blob);

    const link = document.createElement("a");
    link.download = `${filename}.csv`;
    link.href = url;
    link.click();

    URL.revokeObjectURL(url);
  }, []);

  return { exportAsPNG, exportAsSVG, exportDataAsCSV };
}
```

#### 5.3.1 智能告警降噪

```typescript
// src/app/services/alert-deduplication.ts

interface AlertPattern {
  id: string;
  fingerprint: string;  // 基于告警内容生成
  count: number;
  firstSeen: number;
  lastSeen: number;
  suppressed: boolean;
}

export class AlertDeduplicator {
  private patterns: Map<string, AlertPattern> = new Map();
  private threshold = 3; // 连续出现 3 次才告警
  private windowMs = 5 * 60 * 1000; // 5 分钟窗口

  process(alert: AlertData): AlertData | null {
    const fingerprint = this.generateFingerprint(alert);
    const now = Date.now();
    
    const existing = this.patterns.get(fingerprint);
    
    if (existing) {
      existing.count++;
      existing.lastSeen = now;
      
      // 如果在窗口内重复出现，抑制告警
      if (existing.count > this.threshold && 
          now - existing.firstSeen < this.windowMs) {
        existing.suppressed = true;
        return null;
      }
    } else {
      this.patterns.set(fingerprint, {
        id: `pat-${Date.now()}`,
        fingerprint,
        count: 1,
        firstSeen: now,
        lastSeen: now,
        suppressed: false,
      });
    }

    return alert;
  }

  private generateFingerprint(alert: AlertData): string {
    // 基于告警来源、级别、消息关键词生成指纹
    const keywords = alert.message.toLowerCase()
      .split(/\s+/)
      .filter(w => w.length > 3)
      .slice(0, 5)
      .join("-");
    
    return `${alert.source}:${alert.level}:${keywords}`;
  }

  // 清理过期模式
  cleanup() {
    const now = Date.now();
    for (const [key, pattern] of this.patterns) {
      if (now - pattern.lastSeen > this.windowMs * 2) {
        this.patterns.delete(key);
      }
    }
  }
}
```
