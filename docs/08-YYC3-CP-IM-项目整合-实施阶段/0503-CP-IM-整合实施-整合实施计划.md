### 1. 安全功能增强 🔒

#### 1.1 添加内容安全策略 (CSP) 检测

**目标**：检测当前页面的 CSP 配置，验证内联脚本是否被阻止，提供优化建议

**实现步骤**：

1. **创建 Hook**
   ```typescript
   // src/app/hooks/useCSPChecker.ts
   export interface CSPReport {
     cspEnabled: boolean;
     unsafeInline: boolean;
     unsafeEval: boolean;
     scriptSrc: string[];
     styleSrc: string[];
     recommendations: string[];
   }
   
   export function useCSPChecker() {
     const [report, setReport] = useState<CSPReport | null>(null);
     
     const checkCSP = useCallback(() => {
       const meta = document.querySelector('meta[http-equiv="Content-Security-Policy"]');
       const cspHeader = meta?.getAttribute('content') || '';
       
       // 解析 CSP 策略
       const policies = cspHeader.split(';').map(p => p.trim());
       const scriptSrc = policies
         .find(p => p.startsWith('script-src'))
         ?.replace('script-src', '') || '';
       
       const unsafeInline = scriptSrc.includes('unsafe-inline');
       const unsafeEval = scriptSrc.includes('unsafe-eval');
       
       setReport({
         cspEnabled: cspHeader.length > 0,
         unsafeInline,
         unsafeEval,
         scriptSrc: scriptSrc.split(' ').filter(Boolean),
         styleSrc: [],
         recommendations: [
           unsafeInline ? '移除 unsafe-inline，使用 nonce 或 hash' : null,
           unsafeEval ? '移除 unsafe-eval，避免动态脚本执行' : null,
           cspHeader.length === 0 ? '建议启用 CSP 策略' : null,
         ].filter(Boolean) as string[],
       });
     }, []);
     
     useEffect(() => { checkCSP(); }, []);
     
     return { report, checkCSP };
   }
   ```

2. **创建组件**
   ```typescript
   // src/app/components/CSPChecker.tsx
   export function CSPChecker() {
     const { report, checkCSP } = useCSPChecker();
     // UI 实现
   }
   ```

#### 1.2 添加 Cookie 安全检查

**目标**：检查 Secure、HttpOnly、SameSite 标志

**实现步骤**：

1. **创建 Hook**
   ```typescript
   // src/app/hooks/useCookieChecker.ts
   export interface CookieReport {
     cookies: Array<{
       name: string;
       secure: boolean;
       httpOnly: boolean;
       sameSite: string;
       score: number;
     }>;
     recommendations: string[];
   }
   
   export function useCookieChecker() {
     const [report, setReport] = useState<CookieReport>({ cookies: [], recommendations: [] });
     
     const checkCookies = useCallback(() => {
       const cookies = document.cookie.split(';').map(c => {
         const [name, value] = c.trim().split('=');
         return { name, value };
       });
       
       const cookieReports = cookies.map(c => ({
         ...c,
         secure: false, // document.cookie 无法访问 secure 标志
         httpOnly: false, // document.cookie 无法访问 httpOnly 标志
         sameSite: 'unknown',
         score: 50,
       }));
       
       setReport({
         cookies: cookieReports,
         recommendations: [
           '建议在服务端设置 Secure 标志',
           '建议在服务端设置 HttpOnly 标志',
           '建议设置 SameSite=Strict 或 Lax',
         ],
       });
     }, []);
     
     return { report, checkCookies };
   }
   ```

**优先级**：🔴 高

**预计工时**：3 小时

#### 1.3 添加敏感信息泄漏检测

**目标**：扫描 localStorage/sessionStorage 中的敏感数据，检测控制台日志中的敏感信息

**实现步骤**：

1. **创建敏感模式库**
   ```typescript
   // src/app/lib/security/sensitive-patterns.ts
   export const SENSITIVE_PATTERNS = [
     /password/i,
     /token/i,
     /api[_-]?key/i,
     /secret/i,
     /authorization/i,
     /session[_-]?id/i,
     /credit[_-]?card/i,
   ];
   
   export function detectSensitiveInfo(data: string): string[] {
     const matches: string[] = [];
     for (const pattern of SENSITIVE_PATTERNS) {
       const match = data.match(pattern);
       if (match) {
         matches.push(match[0]);
       }
     }
     return matches;
   }
   ```

2. **创建检测 Hook**
   ```typescript
   // src/app/hooks/useSensitiveDataChecker.ts
   export function useSensitiveDataChecker() {
     const [leaks, setLeaks] = useState<Array<{ source: string; key: string; matches: string[] }>>([]);
     
     const scan = useCallback(() => {
       const results = [];
       
       // 扫描 localStorage
       for (let i = 0; i < localStorage.length; i++) {
         const key = localStorage.key(i);
         if (!key) continue;
         const value = localStorage.getItem(key) || '';
         const matches = detectSensitiveInfo(key + value);
         if (matches.length > 0) {
           results.push({ source: 'localStorage', key, matches });
         }
       }
       
       // 扫描 sessionStorage
       for (let i = 0; i < sessionStorage.length; i++) {
         const key = sessionStorage.key(i);
         if (!key) continue;
         const value = sessionStorage.getItem(key) || '';
         const matches = detectSensitiveInfo(key + value);
         if (matches.length > 0) {
           results.push({ source: 'sessionStorage', key, matches });
         }
       }
       
       setLeaks(results);
     }, []);
     
     return { leaks, scan, clear: () => setLeaks([]) };
   }
   ```

3. **集成到安全仪表板**

**优先级**：🔴 高

   ```typescript
   export class PerformanceMonitor {
     private resources: ResourceEntry[] = [];
     
     private observeResources() {
       const observer = new PerformanceObserver((list) => {
         for (const entry of list.getEntries()) {
           if (entry.entryType === 'resource') {
             const resource = entry as PerformanceResourceTiming;
             this.resources.push({
               name: resource.name,
               type: this.getResourceType(resource.name),
               size: resource.transferSize,
               duration: resource.duration,
               cached: resource.transferSize === 0,
             });
           }
         }
       });
       
       observer.observe({ entryTypes: ['resource'] });
     }
     
     private getResourceType(url: string): ResourceEntry['type'] {
       if (url.endsWith('.js')) return 'script';
       if (url.endsWith('.css')) return 'style';
       if (/\.(png|jpg|jpeg|gif|webp|svg)/i.test(url)) return 'image';
       if (/\.(woff|woff2|ttf|eot)/i.test(url)) return 'font';
       return 'other';
     }
     
     getResources() { return this.resources; }
   }
   ```

   ```typescript
       if (diff < -0.1) return 'down';

     }
   ```

   ```typescript
   export function useDataSync<T>(key: string, defaultValue: T) {
     const [data, setData] = useState<T>(defaultValue);
     const channel = new BroadcastChannel(`yyc3_${key}`);
     
     useEffect(() => {
       const loadData = () => {
         const stored = localStorage.getItem(key);
         if (stored) {
           setData(JSON.parse(stored));
         }
       };
       
       loadData();
       channel.addEventListener('message', (e) => {
         if (e.data.key === key) {
           setData(e.data.value);
         }
       });
       
       return () => channel.close();
     }, [key]);
     
     const updateData = useCallback((value: T) => {
       localStorage.setItem(key, JSON.stringify(value));
       channel.postMessage({ key, value });
       setData(value);
     }, [key]);
     
     const exportData = useCallback(() => {
       const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' });
       const url = URL.createObjectURL(blob);
       const a = document.createElement('a');
       a.href = url;
       a.download = `yyc3-${key}-${Date.now()}.json`;
       a.click();
       URL.revokeObjectURL(url);
     }, [data, key]);
     
     const importData = useCallback((file: File) => {
       const reader = new FileReader();
       reader.onload = (e) => {
         try {
           const imported = JSON.parse(e.target?.result as string);
           updateData(imported);
         } catch (error) {
           console.error('Import failed:', error);
         }
       };
       reader.readAsText(file);
     }, [updateData]);
     
     return { data, updateData, exportData, importData };
   ```

   ```typescript
   export function useDataCleaner() {
     const [stats, setStats] = useState({
       localStorage: { count: 0, size: 0 },
       sessionStorage: { count: 0, size: 0 },
       indexedDB: { count: 0, size: 0 },
     });
     
     const calculateStats = useCallback(async () => {
       let lsSize = 0;
       let lsCount = 0;
       for (let i = 0; i < localStorage.length; i++) {
         const key = localStorage.key(i);
         if (key) {
           lsCount++;
           lsSize += (localStorage.getItem(key) || '').length;
         }
       }
       
       setStats({
         localStorage: { count: lsCount, size: lsSize },
         sessionStorage: { count: sessionStorage.length, size: 0 },
         indexedDB: { count: 0, size: 0 },
       });
     }, []);
     
     const clearOldData = useCallback((olderThanDays: number) => {
       const cutoff = Date.now() - olderThanDays * 86400000;
       const keysToDelete: string[] = [];
       
       for (let i = 0; i < localStorage.length; i++) {
         const key = localStorage.key(i);
         if (key && key.endsWith('_timestamp')) {
           const timestamp = parseInt(localStorage.getItem(key) || '0');
           if (timestamp < cutoff) {
             keysToDelete.push(key);
           }
         }
       }
       
       keysToDelete.forEach(key => localStorage.removeItem(key));
       calculateStats();
     }, [calculateStats]);
     
     const clearCache = useCallback(async () => {
       if ('caches' in window) {
         const cacheNames = await caches.keys();
         await Promise.all(cacheNames.map(name => caches.delete(name)));
       }
       calculateStats();
     }, [calculateStats]);
     
     return { stats, calculateStats, clearOldData, clearCache };
   }
   ```
