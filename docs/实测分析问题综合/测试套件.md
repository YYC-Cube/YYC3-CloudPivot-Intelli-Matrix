### 4.3 测试最佳实践

#### 1. 测试命名规范

```typescript
// ✅ 好的命名
describe("ComponentName - 功能描述", () => {
  it("应该做什么", () => {
    // 清晰描述测试目的
  });
});

// ❌ 不好的命名
describe("ComponentName", () => {
  it("test1", () => {
    // 模糊的测试名称
  });
});
```

#### 2. 测试结构规范

```typescript
describe("ComponentName", () => {
  // 1. 初始化
  beforeEach(() => {
    // 设置测试环境
  });

  // 2. 测试分组
  describe("功能分组", () => {
    it("测试用例", () => {
      // 具体测试逻辑
    });
  });

  // 3. 清理
  afterEach(() => {
    // 清理测试环境
  });
});
```

#### 3. 断言规范

```typescript
// ✅ 好的断言
expect(element).toBeInTheDocument();
expect(element).toHaveText("预期文本");
expect(element).toBeDisabled();
expect(mockFunction).toHaveBeenCalledWith(expectedArgs);

// ❌ 不好的断言
expect(element).toBeTruthy();
expect(mockFunction).toHaveBeenCalled();
```

#### 4. Mock 规范

```typescript
// ✅ 好的 Mock
vi.mock("react-router", () => ({
  useNavigate: vi.fn(),
  useLocation: vi.fn(),
}));

// ❌ 不好的 Mock
vi.mock("react-router", () => ({
  useNavigate: () => {},
  useLocation: () => ({}),
}));
```
